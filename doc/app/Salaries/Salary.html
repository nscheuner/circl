def self.interval(from, to)
  raise ArgumentError, "'from' date is missing.".inspect if from.nil?
  raise ArgumentError, "'to' date is missing.".inspect if to.nil?

  where('salaries.from >= ? AND salaries.to <= ? AND is_template = false', from, to)
end

def as_json(options = nil)
  h = super(options)

  h[:created_at] = created_at.to_date

  h[:template_title] = template.try(:title)

  h[:married] = married?
  h[:yearly_salary] = yearly_salary.to_f
  h[:children_count] = children_count

  h[:gross_pay]   = gross_pay.to_f
  h[:net_salary]  = net_salary.to_f
  h[:person_name] = person.name

  h[:items] = items.map(&:as_json)
  h[:tax_data] = selected_tax_data.map(&:as_json)

  h[:employer_value_total]   = employer_value_total.to_f
  h[:employer_percent_total] = employer_percent_total.to_f
  h[:employee_value_total]   = employee_value_total.to_f
  h[:employee_percent_total] = employee_percent_total.to_f

  h[:brut_account]      = brut_account
  h[:net_account]       = net_account
  h[:employer_account]  = employer_account

  h[:summary] = summary

  h[:errors] = errors
  h
end

def create_missing_tax_data
  next_pos = tax_data.size
  existing_ids = tax_data.map(&:tax_id)
  Salaries::Tax.all.reject{ |t| existing_ids.include?(t.id) }.each do |tax|
    data = tax_data.build(:tax => tax, :position => next_pos)
    data.reset
    data.save!
    next_pos += 1
  end
end

def init_items_from_template
  self.items.destroy_all

def pdf_up_to_date?
  return false unless pdf_updated_at

  # Check if pdf requires an update because salary is newer
  return false if updated_at > pdf_updated_at.to_datetime

  # Check if pdf requires an update because its template is newer
  salary_template.updated_at < pdf_updated_at.to_datetime
end

def summary
  h = items.each_with_object(Hash.new(0)) do |item, h|
    next if item.category.blank? || item.taxes.empty?
    h[item.category] += item.value.to_f
  end

  h[I18n.t("salaries.views.taxes")] = 0 - employee_value_total.to_f # deduction should be negative on display

  items.each_with_object(h) do |item, h|
    next if item.category.blank? || !item.taxes.empty?
    h[item.category] += item.value.to_f
  end

  h.map{ |k, v| { :category => k, :value => v } }
end

def to_html
  html = salary_template.html.dup

  html.gsub!(/#PERSON_ADDRESS/, simple_format(person.full_address))
  html.gsub!(/#PERSON_BANK_INFORMATION/, person.bank_informations)
  html.gsub!(/#PERSON_NAME/, person.name)

  html.gsub!(/#SALARY_CREATED_AT/, created_at.to_s)

  if html.match('#SALARY_DETAILS')
    details_haml = File.read(Rails.root.join('app/views/salaries/details.html.haml'))
    details_html = Haml::Engine.new(details_haml).render(self)
    html.gsub!(/#SALARY_DETAILS/, details_html)
  end

  if html.match('#SALARY_ADVANCED_DETAILS')
    advanced_details_haml = File.read(Rails.root.join('app/views/salaries/advanced_details.html.haml'))
    advanced_details_html = Haml::Engine.new(advanced_details_haml).render(self)
    html.gsub!(/#SALARY_ADVANCED_DETAILS/, advanced_details_html)
  end

  html.gsub!(/#SALARY_FROM/, from.to_s)
  html.gsub!(/#SALARY_ID/, id.to_s)
  html.gsub!(/#SALARY_INTERVAL_IN_DAYS/, interval_in_days.to_s)
  html.gsub!(/#SALARY_TITLE/, title)
  html.gsub!(/#SALARY_TO/, to.to_s)
  html.gsub!(/#SALARY_YEARLY_SALARY/, template.yearly_salary.to_s)

  html
end
